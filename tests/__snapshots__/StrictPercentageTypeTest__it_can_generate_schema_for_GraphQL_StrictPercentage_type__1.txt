"""A strict representation of a percentage."""
type StrictPercentage {
  """Whether the percentage is zero."""
  isZero: Boolean!

  """Whether the percentage is negative."""
  isNegative: Boolean!

  """Whether the percentage is negative or zero."""
  isNegativeOrZero: Boolean!

  """Whether the percentage is positive."""
  isPositive: Boolean!

  """Whether the percentage is positive or zero."""
  isPositiveOrZero: Boolean!

  """Retrieve the percentage as a string representation."""
  toString: String!

  """Retrieve the percentage as a float."""
  toFloat: Float!
}
